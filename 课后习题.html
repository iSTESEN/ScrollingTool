    public static double hexToDecimal(String hex) {
        double res = 0;
        hex = String.valueOf(new StringBuilder(hex).reverse());
        for(int i=0; i<hex.length(); i++) {
            res += hexCharToDecimal(hex.charAt(i)) * Math.pow(16, i);
        }
        return res;
    }


    public static int hexCharToDecimal(char ch) {
        if(ch>='A' && ch <='F') {
            return 10 + (ch - 'A');
        } else {
            return ch - '0';
        }
    }
}

假设密码规则如下：
密码必须至少有8个字符。
密码只能包括字母和数字。
密码必须至少有2个数字。
编写一个程序，提示用户输入密码，如果该密码符合规则就显示“ValidPassword”，否则显示“InvalidPassword”。下面是一些运行示例：

public class FileTest {


    public static void main(String[] args) {
        Scanner in = new Scanner(System.in);
        System.out.print("请输入：");
        System.out.println(check(in.nextLine()));

    }

    public static boolean check(String code) {
        // 检查长度
        if(code.length() < 8) {
            return false;
        }
        int num = 0; // 数字个数
        for(char c : code.toCharArray()) {
            if(!isValidChar(c)) {
                return false;
            }
            if(c>='0' && c<='9') {
                num ++;
            }
        }
        // 检查数字个数
        return num >= 2;
    }

    public static boolean isValidChar(char c) {
        c = Character.toUpperCase(c);
        return (c>='0' && c<='9') || (c>='A' && c<='Z');
    }

}

    public static void main(String[] args) {
        Scanner in = new Scanner(System.in);

        int index = 1;

        for (int i=0; i<5; i++) {
            for(int j=0; j<10; j++) {
                while (true) {
                    if(isSuShu(++index)) {
                        System.out.print(index + "  ");
                        break;
                    }
                }
            }
            System.out.println();
        }



    }

    public static boolean isSuShu(int num) {
        if(num <= 1) {
            return false;
        }
        for(int i=2; i<=Math.sqrt(num); i++) {
            if(num % i == 0) {
                return false;
            }
        }
        return true;
    }
}

class Rectangle {

    private double height;

    public Rectangle() {}
    public Rectangle(double width, double height) {
        this.height = height;
        this.width = width;
    }

    public void setWidth(double width) {
        this.width = width;
    }
    public double getWidth() {
        return this.width;

    public void setHeight(double height) {

        return this.height;
    }

        return height * width;
    }
    public double getPerimeter() {
        return (height + width) * 2;

        int[][] arr = new int[][]{{1,4,5,6}, {3,5,8,9}, {6,7,9,8}, {0,0,0,0}};
        print(arr);
        rotate(arr);
        print(arr);

    public static void rotate(int[][] arr) {
        for(int i=0; i<4; i++) {
            for(int j=i; j<4; j++) {
                int temp = arr[i][j];
                arr[i][j] = arr[j][i];
                arr[j][i] = temp;
            }
        }
    }

    public static void print(int[][] arr) {
        for(int i=0; i<4; i++) {
            for(int j=0; j<4; j++) {
                if(arr[i][j] != 0) System.out.print(arr[i][j] + " ");
            }
            System.out.println();
        }
    }

}

    while ((len = reader.read(buf)) != -1 ) {
        System.out.println(new String(buf, 0, len));
    }
    reader.close();
}
