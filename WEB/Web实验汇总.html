2、根据邮件的重量和用户是否选择加急计算邮费 

根据邮件的重量和用户是否选择加急计算邮费。
计算规则：
重量在1000克以内(包括1000克), 基本费8元。超过1000克的部分，每500克加收超重费4元，不足500克部分按500克计算；如果用户选择加急，多收5元。
在prompt弹窗中输入要寄的物品的重量以及是否要加紧（加急输入yes, 不加急输入no），然后在弹出对话框中显示对应的邮费。

<script>
    var weight = parseFloat(prompt('邮件重量g'))
    console.log(weight);
    
    var choose = 'no'
    while(true) {
        choose = prompt('是否加急')
        if(choose!='yes' && choose!='no') {
            alert('输入有误')
        } else {
            break
        }
    }

    // 计算邮费
    function cal(weight, choose) {
        // 1000内
        if(weight<=1000) {
            return 8 + (choose=='yes'? 5 : 0);
        }
        // 计算超重部分
        weight -= 1000
        var t = Math.ceil(weight/500)
        return 8 + t*4 + (choose=='yes'? 5 : 0);
    }

    alert(cal(weight, choose));

</script>

3、倒计时 

在一些电商网站的活动页上会经常出现折扣商品的倒计时标记，显示离活动结束还剩X天X小时X分X秒。编制一个函数，把促销截止的目标时间作为参数传入，然后在控制器台每间隔1s显示促销结束倒计时。

<h1 id="showpanel" align="center"></h1>

<script>
    // 设置结束时间
    var endTime = '2025-1-1 0:0:0'


    // 检查是否需要补0，返回字符串
    function checkStr(ele) {
        if(ele<10) {
            return `0${ele}`;
        } else {
            return `${ele}`;
        }
    }
    
    // 将目标时间-现在时间，转换为字符串传回
    function makeStr(targetTime) {
        var nowTime = new Date().getTime() // 获得时间戳
        targetTime = new Date(targetTime).getTime()
        // 如果超时
        if(nowTime > targetTime) {
            return '倒计时结束'
        }
        // 得到时间差
        var gapTime = targetTime-nowTime
        // 去除毫秒
        gapTime = Math.floor(gapTime/1000)
        
        var resStr = ''
        // 得到秒
        resStr = checkStr(Math.floor(gapTime%60)) + ' 秒'
        gapTime = Math.floor(gapTime/60)
        
        // 得到分
        resStr = checkStr(Math.floor(gapTime%60)) + ' 分 ' + resStr
        gapTime = Math.floor(gapTime/60)
        // 得到时
        resStr = checkStr(Math.floor(gapTime%24)) + ' 小时 ' + resStr
        gapTime = Math.floor(gapTime/24)
        // 得到天
        resStr = gapTime + ' 天 ' + resStr
        return resStr
    }

    // console.log(makeStr('2025-1-1 0:0:0'));
    window.setInterval(()=>{
        console.log(makeStr(endTime));
        document.getElementById('showpanel').innerText = '离活动结束还剩: ' + makeStr(endTime)
    }, 1000)


</script>

4、输出所有指定元素出现的位置以及次数 

要求在一组字符串中，找到并在控制台输出所有指定元素出现的位置以及次数。字符串为“Hello World, Hello JavaScript”。

<script>
    // 查找元素
    var targetStr = 'lo'
    var searchStr = 'Hello World, Hello JavaScript'

    var res = {
        location: [],
        times: 0
    }

    
    var flag = 0 // 起始游标
    while(flag < searchStr.length) {
        var index = searchStr.indexOf(targetStr, flag)
        if(index == -1) {
            break
        } else {
            res.location.push(index)
            res.times++
            flag = index + targetStr.length
        }
    }
    console.log('查找: ', targetStr);
    
    console.log('所有位置: ', res.location);
    console.log('出现次数: ', res.times);
    
    

</script>
